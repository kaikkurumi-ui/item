## 商品表结构

```sql
CREATE TABLE `item` (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY ,
    item_images varchar(2000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '商品图片，用$符拼接',
    title varchar(100) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '商品标题',
    price decimal NOT NULL COMMENT '商品价格',
    description varchar(2000) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci  NOT NULL  COMMENT '商品详情介绍',
    create_time int unsigned NOT NULL ,
    update_time int unsigned NOT NULL ,
    is_deleted tinyint unsigned NOT NULL DEFAULT 0,
    KEY `idx_name` (`title`)
)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT '商品信息表'
```
